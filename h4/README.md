<a name="delay"></a>
### Fault testing: delay
Delay testing implemented via feature-flag (see [v-service-ff-v2-delay.yaml](k8s/virtual-service/v-service-ff-v2-delay.yaml))
```yaml
apiVersion: networking.istio.io/v1alpha3
kind: VirtualService
metadata:
  name: frontend
  namespace: istio-course
spec:
  hosts:
    - frontend
  http:
    - match:
      - headers:
          developer:
            exact: ssamsonov
      route:
      - destination:
          host: frontend
          subset: v2
    - route:
      - destination:
          host: frontend
          subset: v1

---

apiVersion: networking.istio.io/v1alpha3
kind: VirtualService
metadata:
  name: authors
  namespace: istio-course
spec:
  hosts:
    - authors
  http:
    - match:
        - headers:
            developer:
              exact: ssamsonov
      fault:
        delay:
          fixedDelay: 40s
          percent: 100
      route:
        - destination:
            host: authors
            subset: v2
    - route:
        - destination:
            host: authors
            subset: v1

```
